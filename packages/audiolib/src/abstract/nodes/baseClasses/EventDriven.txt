// EventDrivenNode.ts
import { DefaultEventBus, EventType } from '@/events';
import { NodeID } from '@/types/global';

// Creates a unique event bus for each instance of the class
export abstract class EventDriven extends DefaultEventBus {
  readonly id: NodeID;

  constructor(id: NodeID) {
    super(id);
    this.id = id;
  }

  protected disposeEvents(types: EventType[] = []): void {
    this.clearAllListeners();
    // types.forEach((type) => this.clearListenersForType(type));
  }
}
